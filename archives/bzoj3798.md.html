Description：

求 $[A,B]$ 之间的质数个数，并且满足 $X=Q^2+P^2$ ， $P,Q$ 是正整数。
$1\le A \le B \le3\times 10^8$

Solution：

分块打表，每 $10^5$ 打一个数，对于剩下的暴力。

#include<algorithm>

#include<iostream>

#include<cstdlib>

#include<cstdio>

#include<cmath>

#include<bitset>

#include<cstring>

using namespace std;

#define MAXN 300000005

bitset<MAXN> isprime,isplus;

int prime[MAXN >> 1],sum[MAXN],s[18000];

int tot = 0;

int main()

{

    for(int i = 2;i < MAXN;++i){isprime[i] = true;}

    for(int i = 2;i < MAXN;++i)

    {

        if(isprime[i])prime[++tot] = i;

        for(int j = 1;j <= tot && i * prime[j] < MAXN;++j)

        {

            isprime[i * prime[j]] = false;

            if(i % prime[j] == 0)break;

        }

    }

    for(int i = 1;i < 18000;++i)s[i] = i * i;

    for(int i = 1;i < 18000;++i)

    {

        for(int j = 1;j < 18000;++j)

        {

            if(s[i] + s[j] < MAXN)isplus[s[i] + s[j]] = true;

            else break;

        }

    }

    sum[0] = 0;

    int cnt = 0;

    for(int i = 1;i < MAXN;++i)

    {

        if(isprime[i] && isplus[i])sum[i] = sum[i - 1] + 1;

        else sum[i] = sum[i - 1];

        if((i % 100000) == 0)

		{

			printf("%d,",sum[i]);

			++cnt;

			if(cnt % 11 == 0)puts("");

		}

    }

    return 0;

}

bool check(int k)

{

	if(k == 2)return true;

	int j = 0;

	while(s[j] <= k)++j;

	bool flag = false;

	for(int i = 2;s[i] <= k;++i)

	{

		if(k % i == 0)return false;

		while(s[i] + s[j] > k)--j;

		if(s[i] + s[j] == k)flag = true;

	}

	return flag;

}

	return ans;

}
