$1\le n \le 88888$

#include<algorithm>

#include<iostream>

#include<cstdlib>

#include<cstdio>

#include<cmath>

#include<cstring>

using namespace std;

int n;

#define MAXN 100000

struct opt

{

	int x,y,d;

}o[MAXN];

struct scan

{

	int pos,x,y,id,type;

}s[MAXN << 2];

bool cmp_x(scan a,scan b){return a.x < b.x;}

bool cmp_pos(scan a,scan b){return a.pos < b.pos;}

int cnt = 0;

int ans[MAXN];

#define MAXX 8000000

int c[MAXX];

int lowbit(int x){return (x & (-x));}

void add(int p,int k)

{

	for(int i = p;i < MAXX;i += lowbit(i))c[i] += k;

	return;

}

int query(int p)

{

	int res = 0;

	for(int i = p;i >= 1;i -= lowbit(i))res += c[i];

	return res;

}

void cdq(int l,int r)

{

	if(l == r)return;

	int mid = ((l + r) >> 1);

	cdq(l,mid);cdq(mid + 1,r);

	sort(s + l,s + mid + 1,cmp_x);sort(s + mid + 1,s + r + 1,cmp_x);

	int i = l,j = mid + 1;

	for(;i <= mid && j <= r;)

	{

		if(s[i].x <= s[j].x)

		{

			if(s[i].type == 0)

			{

				add(s[i].y,1);

			}

			++i;

		}

		else

		{

			if(s[j].type != 0)

			{

				ans[s[j].id] += query(s[j].y) * s[j].type;

			}

			++j;

		}

	}

	for(;j <= r;++j)

		if(s[j].type != 0)

			ans[s[j].id] += query(s[j].y) * s[j].type;

	for(int k = l;k < i;++k)

		if(s[k].type == 0)

			add(s[k].y,-1);

	sort(s + l,s + r + 1,cmp_pos);

	return;

}

int main()

{

	memset(c,0,sizeof(c));

	scanf("%d",&n);

	for(int i = 1;i <= n;++i)

	{

		scanf("%d%d%d",&o[i].x,&o[i].y,&o[i].d);

		o[i].x += 3;o[i].y += 3;

	}

	memset(ans,0,sizeof(ans));

	cnt = 0;

	for(int i = 1;i <= n;++i)

	{

		if(o[i].d == 0)

		{

			++cnt;s[cnt] = (scan){cnt,o[i].x,o[i].y,0,0};

		}

		else

		{

			++cnt;s[cnt] = (scan){cnt,o[i].x - 1,o[i].y - 1,i,1};

			++cnt;s[cnt] = (scan){cnt,o[i].x + o[i].d,o[i].y - 1,i,-1};

		}

	}

	cdq(1,cnt);

	cnt = 0;

	for(int i = 1;i <= n;++i)

	{

		if(o[i].d == 0)

		{

			++cnt;s[cnt] = (scan){cnt,o[i].x,o[i].x + o[i].y,0,0};

		}

		else

		{

			++cnt;s[cnt] = (scan){cnt,o[i].x - 1,o[i].x + o[i].y + o[i].d,i,-1};

			++cnt;s[cnt] = (scan){cnt,o[i].x + o[i].d,o[i].x + o[i].d + o[i].y,i,1};

		}

	}

	cdq(1,cnt);

	for(int i = 1;i <= n;++i)

		if(o[i].d != 0)

			printf("%d\n",ans[i]);

	return 0;

}
